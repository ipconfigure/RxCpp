98  //
99  // The scheduler is FIFO for any time T. Since the observer is scheduling
100  // on_next/on_error/oncompleted the timed schedule calls must be resheduled
101  // when they occur to ensure that production happens after on_next/on_error/oncompleted
102  //

104  auto produce_buffer = [localState](const rxsc::schedulable&) {
105  localState->dest.on_next(std::move(localState->chunks.front()));
106  localState->chunks.pop_front();
107  };
108  auto selectedProduce = on_exception(
109  [&](){return localState->coordinator.act(produce_buffer);},
110  localState->dest);
111  if (selectedProduce.empty()) {
112  return;
113  }

115  auto create_buffer = [localState, selectedProduce](const rxsc::schedulable&) {
116  localState->chunks.emplace_back();
117  auto produce_at = localState->expected + localState->period;
118  localState->expected += localState->skip;
119  localState->worker.schedule(produce_at, [localState, selectedProduce](const rxsc::schedulable&) {
120  localState->worker.schedule(selectedProduce.get());
121  });
122  };
123  auto selectedCreate = on_exception(
124  [&](){return localState->coordinator.act(create_buffer);},
125  localState->dest);
126  if (selectedCreate.empty()) {
127  return;
128  }
129

132  state->skip,

134  localState->worker.schedule(selectedCreate.get());
135  });
