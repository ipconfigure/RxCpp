120  void add(const SubscriberFrom& sf, observer_type o) const {
121  trace_activity().connect(sf, o);
122  std::unique_lock<std::mutex> guard(b->state->lock);
123  switch (b->state->current) {
124  case mode::Casting:
125  {
126  if (o.is_subscribed()) {
127  b->completer = std::make_shared<completer_type>(b->state, b->completer, o);
128  ++b->state->generation;
129  }
130  }
131  break;
132  case mode::Completed:
133  {
134  guard.unlock();
135  o.on_completed();
136  return;
137  }
138  break;
139  case mode::Errored:
140  {
141  auto e = b->state->error;
142  guard.unlock();
143  o.on_error(e);
144  return;
145  }
146  break;
147  default:
148  abort();
149  }
