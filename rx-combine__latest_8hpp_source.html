70  template<int Index, class State>
71  void subscribe_one(std::shared_ptr<State> state) const {
72 
73  typedef typename std::tuple_element<Index, tuple_source_type>::type::value_type source_value_type;
74 
75  composite_subscription innercs;
76 
77  // when the out observer is unsubscribed all the
78  // inner subscriptions are unsubscribed as well
79  state->out.add(innercs);
80 
81  auto source = on_exception(
82  [&](){return state->coordinator.in(std::get<Index>(state->source));},
83  state->out);
84  if (source.empty()) {
85  return;
86  }
87 
88  // this subscribe does not share the observer subscription
89  // so that when it is unsubscribed the observer can be called
90  // until the inner subscriptions have finished
91  auto sink = make_subscriber<source_value_type>(
92  state->out,
93  innercs,
94  // on_next
95  [state](source_value_type st) {
96  auto& value = std::get<Index>(state->latest);
97 
98  if (value.empty()) {
99  ++state->valuesSet;
100  }
101 
102  value.reset(st);
103 
104  if (state->valuesSet == sizeof... (ObservableN)) {
105  auto values = rxu::surely(state->latest);
106  auto selectedResult = rxu::apply(values, state->selector);
107  state->out.on_next(selectedResult);
108  }
109  },
110  // on_error
111  [state](std::exception_ptr e) {
112  state->out.on_error(e);
113  },
114  // on_completed
115  [state]() {
116  if (--state->pendingCompletions == 0) {
117  state->out.on_completed();
118  }
119  }
120  );
121  auto selectedSink = on_exception(
122  [&](){return state->coordinator.out(sink);},
123  state->out);
124  if (selectedSink.empty()) {
125  return;
126  }
127  source->subscribe(std::move(selectedSink.get()));
128  }
