96  //
97  // The scheduler is FIFO for any time T. Since the observer is scheduling
98  // on_next/on_error/oncompleted the timed schedule calls must be resheduled
99  // when they occur to ensure that production happens after on_next/on_error/oncompleted

102  auto release_window = [localState](const rxsc::schedulable&) {
103  localState->worker.schedule([localState](const rxsc::schedulable&) {
104  localState->subj[0].get_subscriber().on_completed();
105  localState->subj.pop_front();
106  });
107  };
108  auto selectedRelease = on_exception(

115  auto create_window = [localState, selectedRelease](const rxsc::schedulable&) {
116  localState->subj.push_back(rxcpp::subjects::subject<T>());
117  localState->dest.on_next(localState->subj[localState->subj.size() - 1].get_observable().as_dynamic());
118 
119  auto produce_at = localState->expected + localState->period;
120  localState->expected += localState->skip;
121  localState->worker.schedule(produce_at, [localState, selectedRelease](const rxsc::schedulable&) {
122  localState->worker.schedule(selectedRelease.get());
123  });
124  };
