187  template<class Obsvbl, class... ArgN>
188  static auto blocking_subscribe(const Obsvbl& source, ArgN&&... an)
189  -> composite_subscription {
190  std::mutex lock;
191  std::condition_variable wake;
192  composite_subscription cs;
193  struct tracking
194  {
195  ~tracking()
196  {
197  if (!disposed || !wakened) abort();
198  }
199  tracking()
200  {
201  disposed = false;
202  wakened = false;
203  false_wakes = 0;
204  true_wakes = 0;
205  }
206  std::atomic_bool disposed;
207  std::atomic_bool wakened;
208  std::atomic_int false_wakes;
209  std::atomic_int true_wakes;
210  };
211  auto track = std::make_shared<tracking>();
212 
213  auto scbr = make_subscriber<T>(std::forward<ArgN>(an)...);
214  cs = scbr.get_subscription();
215  cs.add(
216  [&, track](){
217  // OSX geting invalid x86 op if notify_one is after the disposed = true
218  // presumably because the condition_variable may already have been awakened
219  // and is now sitting in a while loop on disposed
220  wake.notify_one();
221  track->disposed = true;
222  });
223  std::unique_lock<std::mutex> guard(lock);
224  source.subscribe(std::move(scbr));
225  wake.wait(guard,
226  [&, track](){
227  // this is really not good.
228  // false wakeups were never followed by true wakeups so..
229 
230  // anyways this gets triggered before disposed is set now so wait.
231  while (!track->disposed) {
232  ++track->false_wakes;
233  }
234  ++track->true_wakes;
235  return true;
236  });
237  track->wakened = true;
238  if (!track->disposed || !track->wakened) abort();
239  return composite_subscription::empty();
240  }
