72  void ensure_processing(std::unique_lock<std::mutex>& guard) const {
73  if (!guard.owns_lock()) {
74  abort();
75  }
76  if (current == mode::Empty) {
77  current = mode::Processing;
78 
79  auto keepAlive = this->shared_from_this();
80 
81  auto drain = [keepAlive, this](const rxsc::schedulable& self){
82  using std::swap;
83  try {
84  if (drain_queue.empty() || !destination.is_subscribed()) {
85  std::unique_lock<std::mutex> guard(lock);
86  if (!destination.is_subscribed() ||
87  (!lifetime.is_subscribed() && queue.empty() && drain_queue.empty())) {
88  current = mode::Disposed;
89  queue_type expired;
90  swap(expired, queue);
91  guard.unlock();
92  lifetime.unsubscribe();
93  destination.unsubscribe();
94  return;
95  }
96  if (drain_queue.empty()) {
97  if (queue.empty()) {
98  current = mode::Empty;
99  return;
100  }
101  swap(queue, drain_queue);
102  }
103  }
104  auto notification = std::move(drain_queue.front());
105  drain_queue.pop();
106  notification->accept(destination);
107  self();
108  } catch(...) {
109  destination.on_error(std::current_exception());
110  std::unique_lock<std::mutex> guard(lock);
111  current = mode::Errored;
112  queue_type expired;
113  swap(expired, queue);
114  }
115  };
116 
117  auto selectedDrain = on_exception(
118  [&](){return coordinator.act(drain);},
119  destination);
120  if (selectedDrain.empty()) {
121  current = mode::Errored;
122  using std::swap;
123  queue_type expired;
124  swap(expired, queue);
125  return;
126  }
127 
128  auto processor = coordinator.get_worker();
129 
130  RXCPP_UNWIND_AUTO([&](){guard.lock();});
131  guard.unlock();
132 
133  processor.schedule(selectedDrain.get());
134  }
135  }
