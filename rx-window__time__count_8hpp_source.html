112  static std::function<void(const rxsc::schedulable&)> release_window(int id, rxsc::scheduler::clock_type::time_point expected, state_type state) {
113  auto release = [id, expected, state](const rxsc::schedulable&) {
114  if (id != state->subj_id)
115  return;
116 
117  state->subj.get_subscriber().on_completed();
118  state->subj = rxcpp::subjects::subject<T>();
119  state->dest.on_next(state->subj.get_observable().as_dynamic());
120  state->cursor = 0;
121  auto new_id = ++state->subj_id;
122  auto produce_time = expected + state->period;
123  state->worker.schedule(produce_time, [new_id, produce_time, state](const rxsc::schedulable&){
124  state->worker.schedule(release_window(new_id, produce_time, state));
125  });
126  };
127  auto selectedRelease = on_exception(
128  [&](){return state->coordinator.act(release);},
129  state->dest);
130  if (selectedRelease.empty()) {
131  return std::function<void(const rxsc::schedulable&)>();
132  }
133 
134  return std::function<void(const rxsc::schedulable&)>(selectedRelease.get());
135  }
