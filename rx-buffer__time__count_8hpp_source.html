112  static std::function<void(const rxsc::schedulable&)> produce_buffer(int id, rxsc::scheduler::clock_type::time_point expected, state_type state) {
113  auto produce = [id, expected, state](const rxsc::schedulable&) {
114  if (id != state->chunk_id)
115  return;
116 
117  state->dest.on_next(state->chunk);
118  state->chunk.resize(0);
119  auto new_id = ++state->chunk_id;
120  auto produce_time = expected + state->period;
121  state->worker.schedule(produce_time, [new_id, produce_time, state](const rxsc::schedulable&){
122  state->worker.schedule(produce_buffer(new_id, produce_time, state));
123  });
124  };

126  auto selectedProduce = on_exception(

128  state->dest);
129  if (selectedProduce.empty()) {

138  auto work = [v, localState](const rxsc::schedulable& self){
139  localState->chunk.push_back(v);
140  if (int(localState->chunk.size()) == localState->count) {
141  produce_buffer(localState->chunk_id, localState->worker.now(), localState)(self);
