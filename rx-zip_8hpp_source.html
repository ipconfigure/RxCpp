69  template<int Index, class State>
70  void subscribe_one(std::shared_ptr<State> state) const {
71 
72  typedef typename std::tuple_element<Index, tuple_source_type>::type::value_type source_value_type;
73 
74  composite_subscription innercs;
75 
76  // when the out observer is unsubscribed all the
77  // inner subscriptions are unsubscribed as well
78  state->out.add(innercs);
79 
80  auto source = on_exception(
81  [&](){return state->coordinator.in(std::get<Index>(state->source));},
82  state->out);
83  if (source.empty()) {
84  return;
85  }
86 
87  // this subscribe does not share the observer subscription
88  // so that when it is unsubscribed the observer can be called
89  // until the inner subscriptions have finished
90  auto sink = make_subscriber<source_value_type>(
91  state->out,
92  innercs,
93  // on_next
94  [state](source_value_type st) {
95  auto& values = std::get<Index>(state->pending);
96  values.push_back(st);
97  if (rxu::apply_to_each(state->pending, rxu::list_not_empty(), rxu::all_values_true())) {
98  auto selectedResult = rxu::apply_to_each(state->pending, rxu::extract_list_front(), state->selector);
99  state->out.on_next(selectedResult);
100  }
101  },
102  // on_error
103  [state](std::exception_ptr e) {
104  state->out.on_error(e);
105  },
106  // on_completed
107  [state]() {
108  if (--state->pendingCompletions == 0) {
109  state->out.on_completed();
110  }
111  }
112  );
113  auto selectedSink = on_exception(
114  [&](){return state->coordinator.out(sink);},
115  state->out);
116  if (selectedSink.empty()) {
117  return;
118  }
119  source->subscribe(std::move(selectedSink.get()));
120  }
