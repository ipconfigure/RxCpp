189  struct seed_type
190  {
191  seed_type()
192  : value()
193  , count(0)
194  {
195  }
196  T value;
197  int count;
198  rxu::detail::maybe<double> stage;
199  };
200  seed_type seed() {
201  return seed_type{};
202  }
203  seed_type operator()(seed_type a, T v) {
204  if (a.count != 0 &&
205  (a.count == std::numeric_limits<int>::max() ||
206  ((v > 0) && (a.value > (std::numeric_limits<T>::max() - v))) ||
207  ((v < 0) && (a.value < (std::numeric_limits<T>::min() - v))))) {
208  // would overflow, calc existing and reset for next batch
209  // this will add error to the final result, but the alternative
210  // is to fail on overflow
211  double avg = a.value / a.count;
212  a.value = v;
213  a.count = 1;
214  if (!a.stage.empty()) {
215  a.stage.reset((*a.stage + avg) / 2);
216  } else {
217  a.stage.reset(avg);
218  }
219  } else {
220  a.value += v;
221  ++a.count;
222  }
223  return a;
224  }
225  double operator()(seed_type a) {
226  if (a.count > 0) {
227  double avg = a.value / a.count;
228  if (!a.stage.empty()) {
229  avg = (*a.stage + avg) / 2;
230  }
231  return avg;
232  }
233  return a.value;
234  }
235 };
